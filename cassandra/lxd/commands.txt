lxc launch images:ubuntu/20.04 cassandra-node1
lxc launch images:ubuntu/20.04 cassandra-node2
lxc launch images:ubuntu/20.04 cassandra-node3
sudo lxc list


sudo lxc config set cassandra-node1 limits.memory 2GB
sudo lxc config set cassandra-node1 limits.cpu.allowance 20%
sudo lxc config show cassandra-node1

sudo lxc config set cassandra-node2 limits.memory 2GB
sudo lxc config set cassandra-node2 limits.cpu.allowance 20%
sudo lxc config show cassandra-node2

sudo lxc config set cassandra-node3 limits.memory 2GB
sudo lxc config set cassandra-node3 limits.cpu.allowance 20%
sudo lxc config show cassandra-node3

lxc file push --create-dirs --uid 0 --gid 0 ~/.ssh/id_rsa.pub cassandra-node1/root/.ssh/authorized_keys
lxc file push --create-dirs --uid 0 --gid 0 ~/.ssh/id_rsa.pub cassandra-node2/root/.ssh/authorized_keys
lxc file push --create-dirs --uid 0 --gid 0 ~/.ssh/id_rsa.pub cassandra-node3/root/.ssh/authorized_keys

sudo lxc exec cassandra-node1 bash



echo 'PubkeyAuthentication yes' >> '/etc/ssh/sshd_config
